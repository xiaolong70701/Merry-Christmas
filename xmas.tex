	\noindent \textbf{Merry Christmas Proof}
	
	\begin{proof}
	Given a function as follow:
	\[
	y = \frac{\log_e\left(\frac{x}{m} -sa\right)}{r^2}
	\]
	By rearranging the equation, we can obtain:
		\begin{align*}
			yr^2 &= \log_e \left(\frac{x}{m} -sa\right)\\
			e^{yr^2} &= \frac{x}{m} -sa\\
			me^{yr^2} &= x - msa\\
			me^{rry} &= x - mas
		\end{align*}
	\end{proof}
